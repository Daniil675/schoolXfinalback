CREATE TABLE users (
id INT AUTO_INCREMENT PRIMARY KEY,
name VARCHAR(63),
email VARCHAR(127),
phone VARCHAR(127),
password VARCHAR(127),
created TIMESTAMP
)
CREATE TABLE images (
id INT AUTO_INCREMENT PRIMARY KEY,
owner_id INT,
name VARCHAR(255),
created TIMESTAMP,
FOREIGN KEY (owner_id)  REFERENCES users(id)
)
CREATE TABLE favorites (
user_id INT,
image_id INT,
created TIMESTAMP,
FOREIGN KEY (user_id)  REFERENCES users(id),
FOREIGN KEY (image_id)  REFERENCES images(id),
PRIMARY KEY(user_id, image_id)
)
CREATE TABLE files (
id INT AUTO_INCREMENT PRIMARY KEY,
image_id INT,
size VARCHAR(63),
path VARCHAR(255),
created TIMESTAMP,
FOREIGN KEY (image_id)  REFERENCES images(id)
)
